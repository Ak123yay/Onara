-- ============================================================
-- SECTION 1: CUSTOM TYPES
-- ============================================================

CREATE TYPE plan_type AS ENUM ('free', 'starter', 'pro');

CREATE TYPE site_status AS ENUM (
  'queued',
  'generating',
  'deploying',
  'live',
  'failed',
  'suspended'
);

CREATE TYPE agent_phase AS ENUM (
  'analyst',
  'content_writer',
  'style_agent',
  'planner',
  'prompt_engineer',
  'code_generator',
  'debugger',
  'seo_agent',
  'qa_agent',
  'mobile_agent',
  'deploying',
  'done',
  'error'
);

CREATE TYPE revision_type AS ENUM (
  'initial_generation',
  'user_revision',
  'system_refresh'
);


-- ============================================================
-- SECTION 2: CORE TABLES
-- ============================================================

-- ----------------------------------------------------------
-- USERS
-- Extended profile on top of Supabase auth.users
-- ----------------------------------------------------------
CREATE TABLE public.users (
  id                  UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email               TEXT NOT NULL UNIQUE,
  full_name           TEXT,
  avatar_url          TEXT,

  -- Plan & billing
  plan                plan_type NOT NULL DEFAULT 'free',
  stripe_customer_id  TEXT UNIQUE,
  stripe_subscription_id TEXT UNIQUE,
  subscription_status TEXT,                          -- active, past_due, canceled, etc.

  -- Reverse trial
  is_trial            BOOLEAN NOT NULL DEFAULT TRUE,
  trial_ends_at       TIMESTAMPTZ NOT NULL DEFAULT (NOW() + INTERVAL '14 days'),

  -- Revision tracking (free tier resets 1st of month, paid resets on billing anniversary)
  revisions_used      INT NOT NULL DEFAULT 0,
  revisions_limit     INT NOT NULL DEFAULT 3,        -- 3 free / 10 starter / unlimited pro (-1)
  revisions_reset_at  TIMESTAMPTZ NOT NULL DEFAULT (DATE_TRUNC('month', NOW()) + INTERVAL '1 month'),

  -- Flags
  show_url            BOOLEAN NOT NULL DEFAULT TRUE, -- false when plan=free after trial
  onboarding_complete BOOLEAN NOT NULL DEFAULT FALSE,

  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ----------------------------------------------------------
-- PROJECTS (generated websites)
-- ----------------------------------------------------------
CREATE TABLE public.projects (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id             UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,

  -- Business data imported from Google
  business_name       TEXT NOT NULL,
  business_address    TEXT,
  business_phone      TEXT,
  business_email      TEXT,
  business_website    TEXT,
  business_hours      JSONB,                         -- { monday: "9am-5pm", ... }
  business_photos     TEXT[],                        -- array of Google photo URLs
  business_category   TEXT,                          -- e.g. "Plumber", "HVAC"
  google_place_id     TEXT,
  google_rating       NUMERIC(2,1),
  google_review_count INT,

  -- Style preferences (collected in build flow Step 13)
  style_preferences   JSONB,                         -- { colorScheme, tone, layout }

  -- Generation output
  status              site_status NOT NULL DEFAULT 'queued',
  current_agent       agent_phase,
  error_message       TEXT,
  generation_ms       INT,                           -- total pipeline duration in ms

  -- Deployment
  cloudflare_project_name TEXT UNIQUE,               -- {projectId} used in pages.dev subdomain
  public_url          TEXT,                          -- https://{projectId}.pages.dev
  custom_domain       TEXT,                          -- user-connected domain, if any
  github_path         TEXT,                          -- sites/{projectId}/ in onara-sites repo

  -- Supabase storage
  storage_bucket_path TEXT,                          -- backup path in Supabase bucket

  -- SEO metadata generated by Agent 8
  seo_title           TEXT,
  seo_description     TEXT,
  seo_keywords        TEXT[],

  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  last_deployed_at    TIMESTAMPTZ
);

-- ----------------------------------------------------------
-- PIPELINE JOBS
-- Tracks each generation/revision run
-- ----------------------------------------------------------
CREATE TABLE public.pipeline_jobs (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id          UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
  user_id             UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,

  job_type            revision_type NOT NULL DEFAULT 'initial_generation',
  phase               agent_phase NOT NULL DEFAULT 'analyst',
  status              TEXT NOT NULL DEFAULT 'queued',  -- queued, running, done, failed

  -- Progress (updated as each agent completes)
  agents_completed    INT NOT NULL DEFAULT 0,
  agents_total        INT NOT NULL DEFAULT 10,
  progress_log        JSONB[],                        -- array of { agent, status, ms, timestamp }

  -- Timing
  queued_at           TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  started_at          TIMESTAMPTZ,
  completed_at        TIMESTAMPTZ,
  duration_ms         INT,

  -- Error detail
  error_agent         agent_phase,
  error_message       TEXT,
  retry_count         INT NOT NULL DEFAULT 0,

  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ----------------------------------------------------------
-- REVISIONS
-- Each user-requested change to an existing site
-- ----------------------------------------------------------
CREATE TABLE public.revisions (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id          UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
  user_id             UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  pipeline_job_id     UUID REFERENCES public.pipeline_jobs(id),

  instruction         TEXT NOT NULL,                 -- plain-language change request
  affected_components TEXT[],                        -- which atomic components were regenerated
  status              TEXT NOT NULL DEFAULT 'pending', -- pending, running, done, failed

  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  completed_at        TIMESTAMPTZ
);

-- ----------------------------------------------------------
-- PIPELINE ERRORS
-- Structured error log for production debugging (Step 27.10)
-- ----------------------------------------------------------
CREATE TABLE public.pipeline_errors (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id              UUID REFERENCES public.pipeline_jobs(id),
  user_id             UUID REFERENCES public.users(id),
  project_id          UUID REFERENCES public.projects(id),

  active_agent        agent_phase,
  error_type          TEXT,                          -- RateLimitError, TimeoutError, ParseError, etc.
  error_message       TEXT NOT NULL,
  blackboard_snapshot JSONB,                         -- partial state, excluding large code fields

  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ----------------------------------------------------------
-- GBP SYNC LOG
-- Tracks Google Business Profile change detection (v2.5)
-- ----------------------------------------------------------
CREATE TABLE public.gbp_sync_log (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id          UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,

  field_changed       TEXT NOT NULL,                 -- 'hours', 'phone', 'address', etc.
  old_value           TEXT,
  new_value           TEXT,
  auto_deployed       BOOLEAN NOT NULL DEFAULT FALSE,
  notification_sent   BOOLEAN NOT NULL DEFAULT FALSE,

  detected_at         TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ----------------------------------------------------------
-- STRIPE EVENTS
-- Raw Stripe webhook events for idempotency
-- ----------------------------------------------------------
CREATE TABLE public.stripe_events (
  id                  TEXT PRIMARY KEY,              -- Stripe event ID (evt_...)
  type                TEXT NOT NULL,
  data                JSONB NOT NULL,
  processed           BOOLEAN NOT NULL DEFAULT FALSE,
  processed_at        TIMESTAMPTZ,
  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- USERS: each user can only see and edit their own row
CREATE POLICY "users_select_own" ON public.users
  FOR SELECT USING (auth.uid() = id);

CREATE POLICY "users_update_own" ON public.users
  FOR UPDATE USING (auth.uid() = id);

-- PROJECTS: users can CRUD their own projects only
CREATE POLICY "projects_select_own" ON public.projects
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "projects_insert_own" ON public.projects
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "projects_update_own" ON public.projects
  FOR UPDATE USING (auth.uid() = user_id);

CREATE POLICY "projects_delete_own" ON public.projects
  FOR DELETE USING (auth.uid() = user_id);

-- PIPELINE JOBS: read-only for users (pipeline server writes via service role)
CREATE POLICY "jobs_select_own" ON public.pipeline_jobs
  FOR SELECT USING (auth.uid() = user_id);

-- REVISIONS: users can read and insert their own
CREATE POLICY "revisions_select_own" ON public.revisions
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "revisions_insert_own" ON public.revisions
  FOR INSERT WITH CHECK (auth.uid() = user_id);

-- PIPELINE ERRORS: users can see errors for their own jobs
CREATE POLICY "errors_select_own" ON public.pipeline_errors
  FOR SELECT USING (auth.uid() = user_id);

-- GBP SYNC LOG: users can read their own project's sync history
CREATE POLICY "gbp_select_own" ON public.gbp_sync_log
  FOR SELECT USING (
    auth.uid() = (SELECT user_id FROM public.projects WHERE id = project_id)
  );

-- STRIPE EVENTS: no direct user access — service role only
CREATE POLICY "stripe_events_deny_all" ON public.stripe_events
  FOR ALL USING (FALSE);

-- Auto-update updated_at timestamp
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$;

CREATE TRIGGER users_updated_at
  BEFORE UPDATE ON public.users
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER projects_updated_at
  BEFORE UPDATE ON public.projects
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- Auto-create user profile row on Supabase auth signup
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  INSERT INTO public.users (id, email, full_name, avatar_url)
  VALUES (
    NEW.id,
    NEW.email,
    NEW.raw_user_meta_data->>'full_name',
    NEW.raw_user_meta_data->>'avatar_url'
  );
  RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

-- Enforce project limits per plan
CREATE OR REPLACE FUNCTION enforce_project_limit()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE
  current_count INT;
  max_allowed INT;
  user_plan plan_type;
BEGIN
  SELECT plan INTO user_plan FROM public.users WHERE id = NEW.user_id;

  current_count := (
    SELECT COUNT(*) FROM public.projects
    WHERE user_id = NEW.user_id AND status != 'failed'
  );

  max_allowed := CASE user_plan
    WHEN 'free'    THEN 1
    WHEN 'starter' THEN 1
    WHEN 'pro'     THEN 3
    ELSE 1
  END;

  IF current_count >= max_allowed THEN
    RAISE EXCEPTION 'Project limit reached for plan %', user_plan;
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER check_project_limit
  BEFORE INSERT ON public.projects
  FOR EACH ROW EXECUTE FUNCTION enforce_project_limit();

-- Decrement revisions_used on new revision
CREATE OR REPLACE FUNCTION decrement_revision()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  UPDATE public.users
  SET revisions_used = revisions_used + 1
  WHERE id = NEW.user_id;
  RETURN NEW;
END;
$$;

CREATE TRIGGER on_revision_created
  AFTER INSERT ON public.revisions
  FOR EACH ROW EXECUTE FUNCTION decrement_revision();


-- ============================================================
-- SECTION 6: pg_cron JOBS
-- Enable pg_cron extension first:
--   CREATE EXTENSION IF NOT EXISTS pg_cron;
--   GRANT USAGE ON SCHEMA cron TO postgres;
-- ============================================================

-- Reset free tier revisions on the 1st of each month at midnight UTC
SELECT cron.schedule(
  'reset-free-revisions',
  '0 0 1 * *',
  $$
    UPDATE public.users
    SET revisions_used = 0,
        revisions_reset_at = DATE_TRUNC('month', NOW()) + INTERVAL '1 month'
    WHERE plan = 'free';
  $$
);

-- Downgrade expired trials daily at 2am UTC
SELECT cron.schedule(
  'downgrade-expired-trials',
  '0 2 * * *',
  $$
    UPDATE public.users
    SET plan = 'free',
        is_trial = FALSE,
        show_url = FALSE,
        revisions_limit = 3
    WHERE is_trial = TRUE
      AND trial_ends_at < NOW();
  $$
);

-- Suspend sites for failed payments daily at 3am UTC
SELECT cron.schedule(
  'suspend-failed-payment-sites',
  '0 3 * * *',
  $$
    UPDATE public.projects p
    SET status = 'suspended'
    FROM public.users u
    WHERE p.user_id = u.id
      AND u.subscription_status = 'past_due'
      AND p.status = 'live';
  $$
);

-- Clean up old pipeline error logs (keep 30 days)
SELECT cron.schedule(
  'cleanup-pipeline-errors',
  '0 4 * * 0',
  $$
    DELETE FROM public.pipeline_errors
    WHERE created_at < NOW() - INTERVAL '30 days';
  $$
);

-- Dashboard summary per user
CREATE OR REPLACE VIEW public.user_dashboard AS
SELECT
  u.id,
  u.email,
  u.plan,
  u.is_trial,
  u.trial_ends_at,
  u.revisions_used,
  u.revisions_limit,
  u.revisions_reset_at,
  COUNT(p.id) AS total_projects,
  COUNT(p.id) FILTER (WHERE p.status = 'live') AS live_projects
FROM public.users u
LEFT JOIN public.projects p ON p.user_id = u.id
GROUP BY u.id;

-- Active pipeline jobs (for queue monitoring)
CREATE OR REPLACE VIEW public.active_jobs AS
SELECT
  j.*,
  p.business_name,
  u.email AS user_email,
  u.plan AS user_plan
FROM public.pipeline_jobs j
JOIN public.projects p ON p.id = j.project_id
JOIN public.users u ON u.id = j.user_id
WHERE j.status IN ('queued', 'running')
ORDER BY j.queued_at ASC;
